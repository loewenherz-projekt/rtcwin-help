dclose

    

RTC Function

Closes a directory.

Declaration:

void dclose (int refno)

Group:

File handling

Parameters:

refno  

Logical reference number

Return value:

-

 

The function dclose closes the directory specified by the parameter refno. refno is the logical reference number that was assigned by the function dopen.

 

Example:

#define MAX_LOG_FILES  99

string LogFiles[100][63];

 

/* Read available log files into a string array.
   Returns number of available files. */. */

int ReadLogFiles (string directory, string ext)

{

int i, log_file_no = 0;

string fname[63] = "";

 

  infobox ("Read files...");

  for (i=0; i<MAX_LOG_FILES; i++)

    LogFiles[i] = "";

 

  if (dopen (0, directory))           // directory = "shared"

  { 

    if (dsort (0, 3, 1))              // Sort names by date/time, oldest first

    {

      while (dread (0, fname) && log_file_no < MAX_LOG_FILES)

      {

        for (i = length (fname) - 1 ; i > 0 ; i--) 

          if (fname[i] == '.') 

            break;

            

        if (sub (fname, i + 1, 9) == ext || ext == "*") 

          LogFiles[log_file_no++] = fname;

      }

      dclose (0);

      return (log_file_no);

    }

    else

    {

      dclose (0);

      return (-2);  // No file found

    }

  }

  else 

    return (-1); 

}
