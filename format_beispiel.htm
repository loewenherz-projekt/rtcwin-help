The Programming Language RTC     Printout Formatting

Example

    

The little sample program below simply waits for the F5 function key to be pressed. If the key is pressed, the scale is read in format for W&M approval and the following ticket sent to the printer.

 

 

The sample program:

 

int Port = 1;

int ScaleNo;

 

form Ticket

{

    1   1 date 

    1  10 time 

    1  16 "Scale" 

    1  22 ScaleNo

    2   1 "Gross"   

    2  10 fgross bold 

    3   1 "Tare"    

    3  10 ftare  bold

    4   1 "Net"

    4  10 <27,87,49> fnet <27,87,48>  /* Expand: ESC W 1/0 */

    8   1 " " 

}

 

void main (void)

{

char ch = 0;

int result = 0;

 

  fctinfo (5, "Print");  // F5 = Print
  for (;;)

  {

    ch = readkey();

    if (ch == fct5)

    { 

      showtext (20, 20, "Reading weight..."); 

      result = getapprovedweight (indicatedscale());  // Read scale settled

      if (result == 0)

      {

        showtext (20, 20, "Printing...          ");

        print (Port, Ticket);                         // Print ticket on channel Port

        showtext (20, 20,"                      ");

      }

    }

    if (ch == fct6) break;

  }

}
