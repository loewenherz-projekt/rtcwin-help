selectpage

    

RTC Function

Selects a page and makes it visible.

Declaration:

void selectpage (int id)

Group:

Touch screen programming

Parameters:

id

Logical reference number of the page

Return value:

-

 

The instruction selectpage selects the page specified by the parameter id and shows it on the screen. The previous screen is hidden automatically.

 

Example:

void WeighingApplication (void)

{

int source, type, ident;

string event_data[4][255];

 

  selectpage (PAGE_WEIGHING_MODE);

  dropevents (); // Drops all unprocessed events.

  for (;;)

  {

    if (eventreceived() > 0) // Get all events...

    {

      readevent (source, type, ident, event_data); 

      if (source == EVENT_CLASS_APPLICATION) // Only events raised by the application...

      {

        switch (type)

        {

          case EVENT_TYPE_BUTTON_PRESS : 

                     switch (ident)

                     {

                       case ID_BUTTON_0 : return; // Pressed Button_0 return to main menu

                                          break;

                       case ID_BUTTON_1 : /* ... */

                                          break;

                     }

                     break;

          

          case EVENT_TYPE_TEXT_ENTERED :

                     /* ... */

                     break;

          /* ... */

        }

      }

    }

    /* ... */

    delay (0.1);

  }

}
