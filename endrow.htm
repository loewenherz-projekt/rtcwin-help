endrow

    

RTC Function

Ends the definition of a row in a grid layout.

Declaration:

void endrow (void)

Group:

Touch screen programming

Parameters:

-

 

Return value:

-

 

Example:

enum

{

  PAGE_1,

  PAGE_2,

}

 

enum

{

  ID_BUTTON_0,

  ID_BUTTON_1,

  ID_BUTTON_2,

  ID_BUTTON_3,

  ID_BUTTON_4,

  ID_BUTTON_5,

}

 

void CreatePage (void) 

{

  begindraw();

  createpage (PAGE_2);

  page_background (PAGE_2, "#d0e0ff");  // SysTec Default

    // Outer layout as grid:

    beginlayout (LAYOUT_TYPE_GRID);

      // First line: three buttons horizontally aligned

      beginrow();

        createwidget (ID_BUTTON_0, WIDGET_TYPE_BUTTON);

        createwidget (ID_BUTTON_1, WIDGET_TYPE_BUTTON);

        createwidget (ID_BUTTON_2, WIDGET_TYPE_BUTTON);

      endrow();  

      // Second line: two buttons horizontally aligned

      beginrow();

        createwidget (ID_BUTTON_3, WIDGET_TYPE_BUTTON);

        createwidget (ID_BUTTON_4, WIDGET_TYPE_BUTTON)

      endrow();  

    endlayout(); // close outer layout  

  enddraw();

}  

 

The instruction endrow ends the definition of a row in a layout of type LAYOUT_TYPE_GRID, started with most recent beginrow instruction.
