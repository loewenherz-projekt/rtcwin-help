windowstyle

    

RTC Function

Sets the policy for the style of a window.

Declaration:

void windowstyle (int id, string style)

Group:

Touch screen programming

Parameters:

id
style

Logical reference number of the window
Format string

Return value:

-

 

The instruction windowstyle sets the criteria for the look of the window specified by the parameter id. The string parameter style contains the formatting instructions in style sheet syntax.

 

Example:

enum

{

  PAGE_1,

  PAGE_2,

  PAGE_3,

  SOFTWARE_VERSION_WINDOW,

  /* .. */

}

 

// Draw Window

#define ST_BOX_STYLE       "background:    #5a7cf9;" \

                           "border-style:  outset;" \ 

                           "border-width:  0.1px;" \

 

// Ouput of the window

#define ST_BOXTEXT_STYLE   "background:    #5a7cf9;"  \

                           "color:         white;" \

                           "padding:       0.3em;"                       \

                           "border-color:  white white white white;"     \

                           "padding-left:  0.5em; padding-right: 0.5em;" \

                           "padding-top:   10px; padding-bottom: 10px;"  \

                           "border-style:  solid;"                       \

                           "border-radius: 5px"

 

void CreateInfoBox (void)         

{

  begindraw ();

  createwindow (SOFTWARE_VERSION_WINDOW, WINDOW_TYPE_POPUP, "");

  windowstyle  (SOFTWARE_VERSION_WINDOW, ST_BOX_STYLE);

    beginlayout (LAYOUT_TYPE_VBOX, 0);

      createwidget (GUI_OUTPUT_STATUS, WIDGET_TYPE_TEXT_OUTPUT);

        widget_size_policy    (GUI_OUTPUT_STATUS,  SIZE_POLICY_MINIMUM, SIZE_POLICY_MINIMUM);

        widget_style          (GUI_OUTPUT_STATUS,  ST_BOXTEXT_STYLE);

        widget_fontsize       (GUI_OUTPUT_STATUS,  30);                                

        widget_foreground     (GUI_OUTPUT_STATUS,  "white");   // white or #FFFFF

        widget_text_alignment (GUI_OUTPUT_STATUS,  (WIDGET_TEXT_ALIGN_HCENTER | WIDGET_TEXT_ALIGN_VCENTER));

    endlayout();

  enddraw ();

}

 

// Show version info for 2 seconds 

void StartUpMessage (void)              

{

  begindraw();

    selectwindow (SOFTWARE_VERSION_WINDOW);

    widget_text (GUI_OUTPUT_STATUS, TX_PROGRAM_VERSION));

    showwindow (SOFTWARE_VERSION_WINDOW);

  enddraw();

  delay (2);

  hidewindow (SOFTWARE_VERSION_WINDOW);

}

 

Note:

The usage of styles requires considerably processor power and may have a negative impact on the performance.
